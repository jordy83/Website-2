  <script>
    async function haalInschrijvingenOp() {
      try {
        const res = await fetch('http://localhost:3000/inschrijvingen');
        const data = await res.json();

        const tbody = document.getElementById('tabelBody');
        const status = document.getElementById('status');

        if (data.length === 0) {
          status.textContent = "Er zijn nog geen inschrijvingen.";
          return;
        }

        data.forEach(item => {
          const rij = document.createElement('tr');
          rij.innerHTML = `
            <td class="border px-4 py-2">${item.naam}</td>
            <td class="border px-4 py-2">${item.klas}</td>
            <td class="border px-4 py-2">${item.toets}</td>
            <td class="border px-4 py-2">${new Date(item.datum).toLocaleString()}</td>
          `;
          tbody.appendChild(rij);
        });
      } catch (err) {
        document.getElementById('status').textContent = "‚ùå Fout bij ophalen van inschrijvingen.";
        console.error(err);
      }
    }

    haalInschrijvingenOp();
  </script>
